#include <Keypad.h>
#include <LiquidCrystal_I2C.h>

//       LCD SETUP 
LiquidCrystal_I2C lcd(0x27, 16, 2);  


//       KEYPAD SETUP 
const byte ROWS = 4;
const byte COLS = 4;

char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {9, 8, 7, 6};
byte colPins[COLS] = {5, 4, 3, 2};

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

//        OUTPUT PINS 
const int greenLED = 11;
const int redLED   = 10;
const int buzzer   = 12;

//        PASSWORD 
String password = "1504";
String input = "";

void setup() {
  pinMode(greenLED, OUTPUT);
  pinMode(redLED, OUTPUT);
  pinMode(buzzer, OUTPUT);

  lcd.init();
  lcd.backlight();

  lcd.setCursor(0, 0);
  lcd.print("Enter Password:");
  lcd.setCursor(0, 1);
}

void loop() {
  char key = keypad.getKey();

  if (key) {
    tone(buzzer, 1000, 100);

    // CLEAR INPUT
    if (key == '*') {
      input = "";
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Enter Password:");
      lcd.setCursor(0, 1);
      digitalWrite(greenLED, LOW);
      digitalWrite(redLED, LOW);
    }

    // SUBMIT PASSWORD
    else if (key == '#') {
      lcd.clear();

      if (input == password) {
        lcd.setCursor(0, 0);
        lcd.print("Access Granted");
        lcd.setCursor(0, 1);
        lcd.print("Welcome");
        digitalWrite(greenLED, HIGH);
        digitalWrite(redLED, LOW);
        tone(buzzer, 2000, 300);
      } else {
        lcd.setCursor(0, 0);
        lcd.print("Access Denied ");
        lcd.setCursor(0, 1);
        lcd.print("Try Again");
        digitalWrite(redLED, HIGH);
        digitalWrite(greenLED, LOW);
        tone(buzzer, 500, 500);
      }

      delay(2000);

      // RESET SYSTEM
      input = "";
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Enter Password:");
      lcd.setCursor(0, 1);
      digitalWrite(greenLED, LOW);
      digitalWrite(redLED, LOW);
    }

    // NORMAL KEY PRESS
    else {
      if (input.length() < 16) {
        input += key;
        lcd.print("*");
      }
    }
  }
}
